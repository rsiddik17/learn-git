1. Git Basics

-- Initialize a repository
    git init
    git clone <url>

-- Check status & logs
    git status
    git log
    git log --oneline --graph

-- Add & save changes
    git add .        # all files
    git add <file>   # specific file
    git commit -m "message"

-- View differences
    git diff
    git diff --staged

-- Undo simple changes
    git restore <file>
    git reset <file>
